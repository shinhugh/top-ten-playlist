Login:
curl http://localhost:8080/api/auth -v -H "Content-Type: application/json" -d '{"name":"test","password":"test"}'

Logout:
curl http://localhost:8080/api/auth -v -X DELETE

Create user account:
curl http://localhost:8080/api/user -v -H "Content-Type: application/json" -d '{"loginName":"test","password":"test","publicName":"Test"}'

Read user account:
curl http://localhost:8080/api/user/dev -v -b "SESSION="

Update user account:
curl http://localhost:8080/api/user/dev -v -X PUT -b "SESSION=" -H "Content-Type: application/json" -d '{"loginName":"test","password":"test","publicName":"Test"}'

Delete user account:
curl http://localhost:8080/api/user/dev -v -X DELETE -b "SESSION="
